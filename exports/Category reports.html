WITH
orders_ytd AS (
  SELECT ID AS order_id
  FROM wp_posts
  WHERE post_type = 'shop_order'
    AND post_status IN ('wc-completed','wc-processing')
    AND post_date >= DATE_FORMAT(CURDATE(), '%Y-01-01')
    AND post_date <  (CURDATE() + INTERVAL 1 DAY)
),
order_lines AS (
  SELECT
    oi.order_id,
    MAX(CASE WHEN oim.meta_key = '_product_id' THEN oim.meta_value END)        AS product_id,
    SUM(CASE WHEN oim.meta_key = '_qty'        THEN oim.meta_value END)        AS qty,
    SUM(CASE WHEN oim.meta_key = '_line_total' THEN oim.meta_value END)        AS line_total
  FROM wp_woocommerce_order_items oi
  JOIN wp_woocommerce_order_itemmeta oim
    ON oim.order_item_id = oi.order_item_id
  WHERE oi.order_item_type = 'line_item'
    AND oi.order_id IN (SELECT order_id FROM orders_ytd)
  GROUP BY oi.order_id, oi.order_item_id
),
refund_lines AS (
  SELECT
    MAX(CASE WHEN oim.meta_key = '_product_id' THEN oim.meta_value END)        AS product_id,
    SUM(CASE WHEN oim.meta_key = '_qty'        THEN oim.meta_value END)        AS qty,
    SUM(CASE WHEN oim.meta_key = '_line_total' THEN oim.meta_value END)        AS line_total
  FROM wp_posts r
  JOIN wp_woocommerce_order_items oi ON oi.order_id = r.ID
  JOIN wp_woocommerce_order_itemmeta oim ON oim.order_item_id = oi.order_item_id
  WHERE r.post_type = 'shop_order_refund'
    AND oi.order_item_type = 'line_item'
    AND r.post_date >= DATE_FORMAT(CURDATE(), '%Y-01-01')
    AND r.post_date <  (CURDATE() + INTERVAL 1 DAY)
  GROUP BY oi.order_item_id
),
cats AS (
  SELECT t.term_id, t.name, tt.parent
  FROM wp_terms t
  JOIN wp_term_taxonomy tt ON tt.term_id = t.term_id
  WHERE tt.taxonomy = 'product_cat'
),
leaf_cats AS (
  SELECT c.term_id
  FROM cats c
  LEFT JOIN cats child ON child.parent = c.term_id
  WHERE child.term_id IS NULL
),
product_primary_cat AS (
  SELECT tr.object_id AS product_id,
         MIN(tt.term_id) AS category_id
  FROM wp_term_relationships tr
  JOIN wp_term_taxonomy tt ON tt.term_taxonomy_id = tr.term_taxonomy_id
  WHERE tt.taxonomy = 'product_cat'
    AND tt.term_id IN (SELECT term_id FROM leaf_cats)
  GROUP BY tr.object_id
),
cat_sales AS (
  SELECT
    ppc.category_id,
    SUM(CAST(ol.line_total AS DECIMAL(18,4))) AS gross_revenue,
    SUM(CAST(ol.qty        AS DECIMAL(18,4))) AS units,
    COUNT(DISTINCT ol.order_id)                AS orders
  FROM order_lines ol
  JOIN product_primary_cat ppc ON ppc.product_id = ol.product_id
  GROUP BY ppc.category_id
),
cat_refunds AS (
  SELECT
    ppc.category_id,
    SUM(CAST(rf.line_total AS DECIMAL(18,4))) AS refunds_amount,
    SUM(CAST(rf.qty        AS DECIMAL(18,4))) AS refunded_units
  FROM refund_lines rf
  JOIN product_primary_cat ppc ON ppc.product_id = rf.product_id
  GROUP BY ppc.category_id
),
cat_totals AS (
  SELECT
    c.term_id                                 AS category_id,
    c.name                                    AS category_name,
    COALESCE(cs.gross_revenue, 0)             AS gross_revenue,
    COALESCE(cr.refunds_amount, 0)            AS refunds_amount,
    (COALESCE(cs.gross_revenue, 0) - COALESCE(cr.refunds_amount, 0)) AS net_revenue,
    (COALESCE(cs.units, 0) - COALESCE(cr.refunded_units, 0))         AS net_units,
    COALESCE(cs.orders, 0)                    AS orders
  FROM cats c
  LEFT JOIN cat_sales cs   ON cs.category_id = c.term_id
  LEFT JOIN cat_refunds cr ON cr.category_id = c.term_id
)
SELECT
  ct.category_id,
  ct.category_name,
  ROUND(ct.net_revenue, 2)                 AS net_revenue_ytd,
  ROUND(ct.gross_revenue, 2)               AS gross_revenue_ytd,
  ROUND(ct.refunds_amount, 2)              AS refunds_ytd,
  CASE WHEN ct.gross_revenue > 0
       THEN ROUND(ct.refunds_amount / ct.gross_revenue, 4)
       ELSE 0 END                          AS refund_rate,
  ct.net_units                             AS units_ytd,
  ct.orders                                AS orders_ytd,
  CASE WHEN ct.net_units > 0
       THEN ROUND(ct.net_revenue / ct.net_units, 2)
       ELSE NULL END                       AS avg_selling_price,
  ROUND(ct.net_revenue / NULLIF(SUM(ct.net_revenue) OVER (), 0), 4)
                                           AS share_of_store_ytd,
  DENSE_RANK() OVER (ORDER BY ct.net_revenue DESC) AS revenue_rank
FROM cat_totals ct
WHERE ct.net_revenue IS NOT NULL
ORDER BY ct.net_revenue DESC
